Una de las caracter√≠sticas m√°s √∫tiles de los lenguajes de programaci√≥n es que permiten que trabajemos con variables. Una variable es un nombre que se refiere a un valor.

from os.path import basename, exists

def download(url):
    filename = basename(url)
    if not exists(filename):
        from urllib.request import urlretrieve

        local, _ = urlretrieve(url, filename)
        print("Downloaded " + str(local))
    return filename
